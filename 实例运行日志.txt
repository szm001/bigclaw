def find_char(string, char):
    """
    查找字符在字符串中的位置

    参数:
        string: 要搜索的字符串
        char: 要查找的字符

    返回:
        字符首次出现的索引，如果未找到则返回 -1
    """
    for i, c in enumerate(string):
        if c == char:
            return i
    return -1


# 测试示例
if __name__ == "__main__":
    test_string = "Hello, World!"
    test_char = "o"
    result = find_char(test_string, test_char)
    print(f"字符 '{test_char}' 在字符串 '{test_string}' 中首次出现的索引: {result}")

    # 测试查找不存在的字符
    test_char2 = "x"
    result2 = find_char(test_string, test_char2)
    print(f"字符 '{test_char2}' 在字符串 '{test_string}' 中首次出现的索引: {result2}")
